ProgressTracker import pandas as pd def process_large_dataset(csv_file_path): """Process large CSV files in batches""" # Initialize components processor = EdgeProcessor(memory_limit=5000) tracker = ProgressTracker() # Read CSV in chunks chunk_size = 1000 csv_reader = pd.read_csv(csv_file_path, chunksize=chunk_size) # Count total chunks for progress tracking total_rows = sum(1 for _ in open(csv_file_path)) - 1 # -1 for header total_chunks = (total_rows // chunk_size) + 1 tracker.start(total_batches=total_chunks) results = [] for i, chunk in enumerate(csv_reader): # Convert to list of dictionaries records = chunk.to_dict('records') # Process batch batch_result = processor.process_batch(records) results.append(batch_result) # Update progress tracker.update(i + 1) print(f"Progress: {tracker.get_percentage():.1f}% - " f"Processed: {batch_result['processed_count']} records") tracker.finish() # Aggregate results total_processed = sum(r['processed_count'] for r in results) total_errors = sum(r['error_count'] for r in results) total_time = sum(r['processing_time'] for r in results) return { 'total_processed': total_processed, 'total_errors': total_errors, 'total_time': total_time, 'average_speed': total_processed / total_time if total_time > 0 else 0 } # Usage result = process_large_dataset('large_dataset.csv') print(f"Processed {result['total_processed']} records in {result['total_time']:.2f}s") print(f"Average speed: {result['average_speed']:.1f} records/second") ``` --- ## üîß Advanced Usage Examples ### Custom Similarity Functions ```python # custom_similarity.py from blueedge.core import register_similarity_function import re def phonetic_similarity(str1, str2): """Custom phonetic similarity using Soundex algorithm""" def soundex(s): """Generate Soundex code for a string""" if not s: return "0000" s = re.sub(r'[^A-Za-z]', '', s.upper()) if not s: return "0000" first_letter = s[0] # Soundex mapping mapping = { 'BFPV': '1', 'CGJKQSXZ': '2', 'DT': '3', 'L': '4', 'MN': '5', 'R': '6' } code = first_letter for char in s[1:]: for key, value in mapping.items(): if char in key: if code[-1] != value: code += value break # Pad or trim to 4 characters code = (code + "000")[:4] return code soundex1 = soundex(str1) soundex2 = soundex(str2) # Calculate similarity based on matching characters matches = sum(1 for a, b in zip(soundex1, soundex2) if a == b) return matches / 4.0 # Register the custom function register_similarity_function('phonetic', phonetic_similarity) # Use in BlueEdge from blueedge import BlueEdge processor = BlueEdge(similarity_function='phonetic') result = processor.compare_names('Smith', 'Smyth') print(f"Phonetic similarity: {result:.2f}") ``` ### Multi-threading for Performance ```python # multithread_example.py from blueedge import BlueEdge import threading import queue import time class MultiThreadProcessor: def __init__(self, num_threads=4): self.num_threads = num_threads self.processors = [BlueEdge() for _ in range(num_threads)] self.input_queue = queue.Queue() self.result_queue = queue.Queue() def worker(self, processor_id): """Worker thread function""" processor = self.processors[processor_id] while True: try: # Get task from queue task = self.input_queue.get(timeout=1) if task is None: # Shutdown signal break # Process the task batch_id, data = task result = processor.clean_data(data) # Put result in output queue self.result_queue.put((batch_id, result)) # Mark task as done self.input_queue.task_done() except queue.Empty: continue def process_parallel(self, data_batches): """Process multiple batches in parallel""" # Start worker threads threads = [] for i in range(self.num_threads): t = threading.Thread(target=self.worker, args=(i,)) t.start() threads.append(t) # Add tasks to queue for i, batch in enumerate(data_batches): self.input_queue.put((i, batch)) # Wait for all tasks to complete self.input_queue.join() # Signal workers to stop for _ in range(self.num_threads): self.input_queue.put(None) # Wait for all threads to finish for t in threads: t.join() # Collect results results = {} while not self.result_queue.empty(): batch_id, result = self.result_queue.get() results[batch_id] = result return results # Usage example processor = MultiThreadProcessor(num_threads=4) # Create sample batches batches = [ [{'first_name': f'User{i}', 'last_name': f'Name{i}'} for i in range(j*100, (j+1)*100)] for j in range(10) # 10 batches of 100 records each ] start_time = time.time() results = processor.process_parallel(batches) end_time = time.time() print(f"Processed {len(batches)} batches in {end_time - start_time:.2f} seconds") print(f"Total records: {sum(r['total_records'] for r in results.values())}") ``` --- ## üß™ Testing Examples ### Unit Testing ```python # test_blueedge.py import unittest from blueedge import BlueEdge from blueedge.core import levenshtein_distance, psutil import os class BenchmarkSuite: def __init__(self): self.processor = BlueEdge() self.results = {} def memory_usage(self): """Get current memory usage in bytes""" process = psutil.Process(os.getpid()) return process.memory_info().rss def benchmark_processing_speed(self, sizes=[100, 500, 1000, 5000]): """Benchmark processing speed across different data sizes""" print("Benchmarking Processing Speed...") for size in sizes: # Generate test data data = [ {'first_name': f'User{i}', 'last_name': f'Name{i}'} for i in range(size) ] # Measure memory before memory_before = self.memory_usage() # Process and measure time start_time = time.time() result = self.processor.clean_data(data) processing_time = time.time() - start_time # Measure memory after memory_after = self.memory_usage() memory_used = memory_after - memory_before # Calculate metrics records_per_second = size / processing_time self.results[f'size_{size}'] = { 'processing_time': processing_time, 'memory_used': memory_used, 'records_per_second': records_per_second, 'accuracy': result['accuracy_score'] } print(f"Size {size}: {processing_time:.3f}s, " f"{records_per_second:.1f} rec/s, " f"{memory_used/1024:.1f}KB") def benchmark_accuracy(self): """Benchmark accuracy across different error types""" print("\nBenchmarking Accuracy...") test_cases = { 'spelling_variations': [ ({'first_name': 'Mohammed', 'last_name': 'Ahmed'}, {'first_name': 'Mohammad', 'last_name': 'Ahmad'}) ], 'honorific_prefixes': [ ({'first_name': 'Dr. Ahmed', 'last_name': 'Hassan'}, {'first_name': 'Ahmed', 'last_name': 'Hassan'}) ], 'abbreviations': [ ({'first_name': 'Robert', 'last_name': 'Smith'}, {'first_name': 'Rob', 'last_name': 'Smith'}) ] } for category, cases in test_cases.items(): correct_detections = 0 total_cases = len(cases) for record1, record2 in cases: result = self.processor.clean_data([record1, record2]) if result['duplicates_count'] > 0: correct_detections += 1 accuracy = (correct_detections / total_cases) * 100 self.results[f'accuracy_{category}'] = accuracy print(f"{category}: {accuracy:.1f}% accuracy") def generate_report(self): """Generate comprehensive benchmark report""" print("\n" + "="*50) print("BLUEEDGE BENCHMARK REPORT") print("="*50) print("\nProcessing Speed Results:") for key, value in self.results.items(): if key.startswith('size_'): size = key.split('_')[1] print(f" {size} records: {value['processing_time']:.3f}s " f"({value['records_per_second']:.1f} rec/s)") print("\nAccuracy Results:") for key, value in self.results.items(): if key.startswith('accuracy_'): category = key.split('_', 1)[1].replace('_', ' ').title() print(f" {category}: {value:.1f}%") print("\nMemory Usage:") memory_values = [v['memory_used'] for k, v in self.results.items() if k.startswith('size_')] if memory_values: avg_memory = sum(memory_values) / len(memory_values) print(f" Average: {avg_memory/1024:.1f}KB per session") # Run benchmark if __name__ == '__main__': benchmark = BenchmarkSuite() benchmark.benchmark_processing_speed() benchmark.benchmark_accuracy() benchmark.generate_report() ``` --- ## üîó Integration Examples ### Django
